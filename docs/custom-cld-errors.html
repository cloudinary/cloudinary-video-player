Cloudinary Video Player

Custom Errors

Full documentation

Example Code:

      
        <video
          id="player"
          controls
          muted
          autoplay
          class="cld-video-player"
          width="500"
        ></video>
        <div class="error-container"></div>
      
      
        var cld = cloudinary.Cloudinary.new({ cloud_name: 'demo_err' });

        var player = cld.videoPlayer('player', {
          publicId: 'snow_horses',
        });
        player.on('error', (e) => {
          if (e.Player.videojs.error_.code === 11) {
            var title = document.querySelector('.error-container');
            title.innerHTML = 'Error !!!!!!!!! ' + e.Player.videojs.error_.message;
            title.style.color = 'red';
          }
        })

        var cld = cloudinary.Cloudinary.new({ cloud_name: 'demo' });

        player = cld.videoPlayer('player', {
          publicId: '360/Golden-Gate-Bridge',
          sourceTypes: ['mp4'],
          transformation: {
            width: 950,
            vc: "auto"
          }
        });
        player.videojs.vr({projection: '360'});

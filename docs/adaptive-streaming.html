        const config =  {cloud_name: 'demo'};

        var playerHls = cloudinary.videoPlayer('example-player-hls', config);
        var playerDash = cloudinary.videoPlayer('example-player-dash' , config);

        playerHls.source(
          'sea_turtle',
          {
            sourceTypes: ['hls'],
            info: {title: 'HLS'}
          }
        );

        playerDash.source(
          'sea_turtle',
          {
            sourceTypes: ['dash'],
            info: {title: 'MPEG-DASH'}
          }
        );

Cloudinary Video Player


Define Interaction Areas manually


      

         <video
          id="player"
          controls
          muted
          class="cld-video-player">
         </video>
      

      
      var cld = cloudinary.Cloudinary.new({cloud_name: 'demo'});
      var options = {fluid: true};

      var interactionAreas = [
        {
          left : 10 ,
          top: 10,
          width: 80,
          height: 40,
          id: 'topSource'
        },
        {
          left : 10 ,
          top: 75,
          width: 80,
          height : 20,
          id: 'topSource'
        }
      ];

      const sources = {
         topSource : 'https://res.cloudinary.com/eran2903/video/upload/w_880,x_800,y_200,c_crop/w_270,eo_8,q_auto:best,f_auto/production_ID_3888264_fuuo7c.mp4',
         bottomSource: 'https://res.cloudinary.com/eran2903/video/upload/w_540,x_1260,y_3100,c_crop/w_270,eo_8,q_auto:best,f_auto/production_ID_3888264_fuuo7c.mp4'
      }

      var player = cld.videoPlayer('manually', options);

      player.source('https://res.cloudinary.com/eran2903/video/upload/eo_8,w_270,q_auto:best,f_auto/production_ID_3888264_fuuo7c.mp4');

      player.addInteractionAreas(interactionAreas, {
          onClick: (event) => {
            const source = sources[event.item.id];
            event.zoom(source);
         }
      });

Interaction Areas using predefined templates

          id="player"

      var cld = cloudinary.Cloudinary.new({cloud_name: 'demo'});

      var options = {fluid: true};

      var player = cld.videoPlayer('player', options);

      var sources = [
        'https://res.cloudinary.com/eran2903/video/upload/w_880,x_800,y_200,c_crop/w_270,eo_8,q_auto:best,f_auto/production_ID_3888264_fuuo7c.mp4',
        'https://www2.hm.com/en_in/productpage.0687948004.html',
        'https://res.cloudinary.com/eran2903/video/upload/w_540,x_1260,y_3100,c_crop/w_270,eo_8,q_auto:best,f_auto/production_ID_3888264_fuuo7c.mp4'
      ]

      player.source('https://res.cloudinary.com/eran2903/video/upload/eo_8,w_270,q_auto:best,f_auto/production_ID_3888264_fuuo7c.mp4', {
        interactionAreas: {
          enable: true,
          templateName: 'portrait', // or landscape/all/center
          onClick: (event) => {
            // 'event.index' is the index of the clicked interaction area
            const src = sources[event.index];
            if(event.index === 1){
                if(window.confirm("Do you like this shirt ?")) {
                    window.open(src, '_blank').focus();
                }
            }
            else {
                event.zoom(src)
            }
          }
        }
      });

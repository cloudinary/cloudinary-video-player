      var cld = cloudinary.Cloudinary.new({cloud_name: 'demo'});
      var options = {fluid: true};

      player = cld.videoPlayer('player', options);

      player.source('https://res.cloudinary.com/eran2903/video/upload/eo_8,w_270,q_auto:best,f_auto/production_ID_3888264_fuuo7c.mp4');

      var tracksItems = [
        {
          left : 10 ,
          top: 10,
          width: 80,
          height: 40,
          src: 'https://res.cloudinary.com/eran2903/video/upload/w_880,x_800,y_200,c_crop/w_270,eo_8,q_auto:best,f_auto/production_ID_3888264_fuuo7c.mp4'
        },
        {
          left : 10 ,
          top: 75,
          width: 80,
          height : 20,
          src: 'https://res.cloudinary.com/eran2903/video/upload/w_540,x_1260,y_3100,c_crop/w_270,eo_8,q_auto:best,f_auto/production_ID_3888264_fuuo7c.mp4'
        }
      ];

      player.addTrackers(tracksItems, {
        onClick: (event) => {
          player.source(tracksItems[event.index].src);
          player.play();
        },
      });

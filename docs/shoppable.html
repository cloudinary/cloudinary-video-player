Example Code:

      
        <video
          id="player"
          playsinline
          controls
          muted
          autoplay
          class="cld-video-player cld-fluid"
        ></video>
      
      
        var cld = cloudinary.Cloudinary.new({ cloud_name: 'demo' });

        // Initialize player
        var player = cld.videoPlayer('player');

        // Define source
        var source = {
          info: {
            title: 'Shoppable Video',
            description: 'Browse products shown in the video.',
          },
          shoppable: {
            products : [
              {
                productName: 'Product 1',
                title: 'Overlay on-hover & Link on-click',
                publicId: 'sample',
                onHover: {
                  action: 'tooltip',
                  args: 'Product 1 Overlay'
                },
                onClick: {
                  action: 'goto',
                  pause: true,
                  args: {
                    url: 'https://cloudinary.com'
                  }
                }
              },
              {
                productName: 'Product 2',
                title: 'Switch image & Seek video',
                publicId: 'woman',
                transformation: {
                  width: 300,
                  height: 200,
                  crop: "fill"
                },
                onHover: {
                  action: 'switch',
                  args: {
                    publicId: 'car_lady_dog',
                    transformation: {
                      width: 300,
                      height: 200,
                      crop: "fill",
                      effect: "grayscale"
                    }
                  }
                },
                onClick: {
                  action: 'seek',
                  pause: false,
                  args: {
                    time: '00:09'
                  }
                }
              },
              {
                productName: 'Product 3',
                publicId: 'sofa_cat',
                title: 'Custom Image Tranformations',
                transformation: {
                  width: 300,
                  height: 300,
                  radius: "150",
                  crop: "fill"
                }
              },
              {
                productName: 'Product 4',
                publicId: 'balloons'
              },
              {
                productName: 'Product 5',
                publicId: 'friends'
              }
            ]
          }
        };

        // Play source
        player.source('snow_horses', source);

        // Initialize player
        var player = cloudinary.videoPlayer('player', { cloud_name: 'demo' });

        // Define source
        var source = { publicId: 'snow_horses', info: { title: 'Snow Horses', subtitle: 'A movie about horses' } };

        // Play source
        player.source(source);

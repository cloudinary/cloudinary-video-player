Example Code:

      

      <video
        id="player-multiple-sources"
        controls
        autoplay
        class="cld-video-player"

      </video>

      
    
      
      cloudinary.videoPlayer('player-multiple-sources', {
        cloudName: 'demo',
        sources: [
          { publicId: 'snow_horses', info: { title: 'Snow Horses', subtitle: 'A movie about snow horses' } },
          { publicId: 'dirt_bike', info: { title: 'Dirt Bike', subtitle: 'A short video of dirt bikes' } },
          {
            publicId: 'marmots',
            label: 'Custom video name',
            info: { title: 'Marmots' },
            textTracks: {
              captions: {
                label: 'Original',
                default: true,
              },
              subtitles: [
                {
                  label: 'English',
                  language: 'en-US',
                },
                {
                  label: 'Polish',
                  language: 'pl-PL',
                },
              ]
            }
          },
        ],
      });
